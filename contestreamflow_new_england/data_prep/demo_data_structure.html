Flow and weather 3-d data frames

#arrays starting w/ "q." contain flow records, 
#                w/ "w." contain weather records (including precip) 

#q.monthly and w.monthly are 3-dimensional arrays
str(q.monthly)
##  num [1:744, 1:125, 1:7] NA NA NA NA NA NA NA NA NA NA ...
##  - attr(*, "dimnames")=List of 3
##   ..$ : chr [1:744] "1949-01-01" "1949-02-01" "1949-03-01" "1949-04-01" ...
##   ..$ : chr [1:125] "01049221" "01049300" "01049396" "01059090" ...
##   ..$ : chr [1:7] "mean" "max" "min" "low" ...
# 1st dimension is dates
#the dates in these matrices are consistent for all sites (and match dates for w.monthly)
#    this also means that for each site, there are records for 
#    each date without gaps, even if the value is NA
head(dimnames(q.monthly)[[1]])
## [1] "1949-01-01" "1949-02-01" "1949-03-01" "1949-04-01" "1949-05-01" "1949-06-01"
# 2nd dimension is USGS gage IDs (site_no)
# these are stored as characters, not numbers, because we need to preserve the leading zeros
head(dimnames(q.monthly)[[2]])
## [1] "01049221" "01049300" "01049396" "01059090" "01063310" "01063452"
# 3rd dimension is flow metrics (or weather metrics for w.monthly)
# mean is monthly mean
# max is the monthly max value of the daily mean.  similar for min
# low is the monthly minimum of the 7-day rolling average
# records.period is the number of daily records that were used to calculate the monthly metric
#    I've kept in months that are missing some daily records, as long as it has >=25 days
#    I've kept track of how many daily values were available, 
#         so it you want be stricter in terms of how many days are missing, 
#         you can set up your own filter to remove monthly records depending the #daily records used
dimnames(q.monthly)[[3]]
## [1] "mean"                   "max"                    "min"                    "low"                   
## [5] "complete"               "records.period"         "records.period.rolling"
